## Quiz

headline: in which cases does # mark the beginning of a comment?

question: |
  #
  1. echo $(( 2#1001 ))
  2. echo This \# won't begin a comment
  3. name=Johnny; echo ${#name}
  4. name=Johnny; echo ${name##Jo}

answers:
  - 1 and 2
  - 1 and 3
  - 1, 3 and 4
  - Neither
